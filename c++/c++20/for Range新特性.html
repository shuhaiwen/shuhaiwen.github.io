# for Range初始化语句

  • 功能：在for的()内进行初始化
  • 示例
#include <iostream>
#include <vector>
int main() {
	for (auto vec = std::vector{ 1, 2, 3 }; auto v : vec) {
		std::cout << v << " ";
	}
}
